permissions

check permissions of given file

Parameters

permissions file :string

Attributes

set:dictionaryset using given file permissions

Returns

  • :null
  • :dictionary

Examples

copy
inspect permissions "bin/arturo"
; [ :dictionary
;     user    :    [ :dictionary
;         read     :        true :boolean
;         write    :        true :boolean
;         execute  :        true :boolean
;     ]
;     group   :    [ :dictionary
;         read     :        true :boolean
;         write    :        false :boolean
;         execute  :        true :boolean
;     ]
;     others  :    [ :dictionary
;         read     :        true :boolean
;         write    :        false :boolean
;         execute  :        true :boolean
;     ]
; ]

copy
permissions.set:#[others:#[write:true]] "bin/arturo"
; gave write permission to 'others'
